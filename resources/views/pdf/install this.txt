// ── AUDIO CONTEXT (lazy init) ────────────────────────────────────────
function getAudioCtx() {
  if (!audioCtx || audioCtx.state === 'closed') {
    // Gemini Live expects 16000Hz INPUT, sends 24000Hz OUTPUT
    audioCtx = new (window.AudioContext || window.webkitAudioContext)();
  }
  if (audioCtx.state === 'suspended') audioCtx.resume();
  return audioCtx;
}

async function startVoice() {
  setStatus('Connecting…');
  try {
    getAudioCtx();
    // Use the pre-initialized aiClient
    liveSession = await aiClient.live.connect({
      model: LIVE_MODEL,
      config: {
        systemInstruction: { parts: [{ text: MIRA_SYSTEM }] },
        generationConfig: {
          responseModalities: ["audio"],
          speechConfig: {
            voiceConfig: { prebuiltVoiceConfig: { voiceName: 'Puck' } }
          }
        }
      },
      callbacks: {
        onopen:   ()    => onSessionOpen(),
        onclose:  ()    => onSessionClose(),
        onerror:  (err) => onSessionError(err),
        onmessage:(msg) => onSessionMessage(msg),
      }
    });
  } catch (err) {
    console.error('Mira session error:', err);
    appendMsg('ai', 'Sorry, I couldn\'t connect.');
    setStatus('Connection failed');
    stopVoice(false);
  }
}

  textSend.disabled = true;

  if (liveSession && isListening) {
    liveSession.sendClientContent({ turns: [{ role: 'user', parts: [{ text }] }] });
    textSend.disabled = false;
    return;
  }

  showTyping();
  try {
    // FIX: Use TEXT_MODEL instead of LIVE_MODEL for generateContent to avoid 404
    const res = await aiClient.models.generateContent({
      model: TEXT_MODEL, 
      contents: [{ role: 'user', parts: [{ text }] }],
      config: {
        systemInstruction: MIRA_SYSTEM + `\n\nShort text replies only.`
      }
    });
    
    removeTyping();
    const reply = res.candidates?.[0]?.content?.parts?.[0]?.text ?? 'Sorry...';
    appendMsg('ai', reply);
    miraSpeakText(reply); 
  } catch(err) {
    removeTyping();
    console.error("Text Error:", err);
    appendMsg('ai', 'Network error. Pakisubukan ulit, okay?');
  }
  textSend.disabled = false;
}

function stopVoice(closeSession = true) {
  isListening = false;
  voiceBtn.classList.remove('listening');
  voiceBtn.querySelector('.icon-mic').style.display  = '';
  voiceBtn.querySelector('.icon-stop').style.display = 'none';
  voiceLabel.textContent = 'Tap mic to talk to Mira';
  visualizer.classList.remove('active');
  cancelAnimationFrame(vizRaf);
  avatarWrap.classList.remove('speaking');
  setStatus('Ready to assist');

  if (scriptProc) { scriptProc.disconnect(); scriptProc = null; }
  if (micStream)  { micStream.getTracks().forEach(t => t.stop()); micStream = null; }
  if (closeSession && liveSession) { try { liveSession.close(); } catch(_){} liveSession = null; }
}

function onSessionClose() { stopVoice(false); }
function onSessionError(err) {

function onSessionMessage(msg) {
  // ── Mira's audio response ──
  const parts = msg.serverContent?.modelTurn?.parts ?? [];
  for (const part of parts) {
    if (part.inlineData?.data) {
      const pcmB64 = part.inlineData.data;
      const mime   = part.inlineData.mimeType ?? 'audio/pcm;rate=24000';
      playQueue.push({ pcmB64, mime });
      drainPlayQueue();
    }
    if (part.text) {
      // Transcript from Mira — show as bubble
      appendMsg('ai', part.text);
    }
  }

  // ── User turn transcript ──
  const userParts = msg.serverContent?.inputTranscription?.parts ?? [];
  for (const up of userParts) {
    if (up.text?.trim()) appendMsg('user', up.text.trim());
  }

  // ── Speaking state ──
  if (msg.serverContent?.generationComplete) {
    avatarWrap.classList.remove('speaking');
    isMiraSpeak = false;
    setStatus('Listening…');
  }
  if (parts.length && parts.some(p => p.inlineData?.data)) {
    avatarWrap.classList.add('speaking');
    isMiraSpeak = true;
    setStatus('Mira is speaking…');
  }
}

// ── MICROPHONE ───────────────────────────────────────────────────────
async function startMicrophone() {
  try {
    micStream = await navigator.mediaDevices.getUserMedia({ audio: true });
    const ctx      = getAudioCtx();
    const source   = ctx.createMediaStreamSource(micStream);
    analyser       = ctx.createAnalyser();
    analyser.fftSize = 64;
    source.connect(analyser);

    // ScriptProcessor to pipe PCM to Gemini
    scriptProc = ctx.createScriptProcessor(4096, 1, 1);
    source.connect(scriptProc);
    scriptProc.connect(ctx.destination);

    scriptProc.onaudioprocess = (e) => {
    if (!liveSession || !isListening) return;
    const float32 = e.inputBuffer.getChannelData(0);
    const pcm16   = float32ToPcm16(float32);
    const b64     = arrayBufferToBase64(pcm16.buffer);
    try {
      liveSession.sendRealtimeInput({
        audio: { data: b64, mimeType: 'audio/pcm;rate=16000' }  // ← 16000 not 24000
      });
    } catch(e) {
      // Session may have closed — stop silently
      if (isListening) stopVoice(false);
    }
  };

    startVisualizer();

  } catch (err) {
    console.error('Microphone error:', err);
    appendMsg('ai', 'Hmm, I can\'t access your microphone. Please allow mic access and try again.');
    stopVoice();
  }
}

// ── VISUALIZER ───────────────────────────────────────────────────────
function startVisualizer() {
  visualizer.classList.add('active');
  const data = new Uint8Array(analyser.frequencyBinCount);
  function frame() {
    vizRaf = requestAnimationFrame(frame);
    analyser.getByteFrequencyData(data);
    bars.forEach((bar, i) => {
      const v = (data[i * 2] || 0) / 255;
      bar.style.height = (6 + v * 22) + 'px';
    });
  }
  frame();
}

// ── AUDIO PLAYBACK (PCM16 → Web Audio) ──────────────────────────────
async function drainPlayQueue() {
  if (playBusy || playQueue.length === 0) return;
  playBusy = true;
  const { pcmB64 } = playQueue.shift();
  await playPcm16(pcmB64);
  playBusy = false;
  drainPlayQueue();
}

async function playPcm16(b64) {
  const ctx    = getAudioCtx();
  const raw    = base64ToArrayBuffer(b64);
  const i16    = new Int16Array(raw);
  const f32    = new Float32Array(i16.length);
  for (let i = 0; i < i16.length; i++) f32[i] = i16[i] / 32768;

  const buffer = ctx.createBuffer(1, f32.length, 24000);  // ← always 24000 for output
  buffer.copyToChannel(f32, 0);

  await new Promise(resolve => {
    const src = ctx.createBufferSource();
    src.buffer = buffer;
    src.connect(ctx.destination);
    src.onended = resolve;
    src.start();
  });
}

function miraSpeakText(text) {
  if (!window.speechSynthesis) return;
  window.speechSynthesis.cancel();

  const utt    = new SpeechSynthesisUtterance(text);
  utt.rate     = 0.92;
  utt.pitch    = 1.05;
  utt.volume   = 1;

  utt.onstart = () => { avatarWrap.classList.add('speaking'); setStatus('Mira is speaking…'); };
  utt.onend   = () => { avatarWrap.classList.remove('speaking'); setStatus('Ready to assist'); };

  function pickVoiceAndSpeak() {
    const voices = window.speechSynthesis.getVoices();
    const preferred = voices.find(v =>
      v.lang === 'fil-PH' ||
      v.name.includes('Samantha') ||
      v.name.includes('Karen') ||
      v.name.includes('Zira') ||
      (v.name.toLowerCase().includes('female'))
    );
    if (preferred) utt.voice = preferred;
    window.speechSynthesis.speak(utt);
  }

  // Voices may not be loaded yet — wait for them
  if (window.speechSynthesis.getVoices().length > 0) {
    pickVoiceAndSpeak();
  } else {
    window.speechSynthesis.onvoiceschanged = () => {
      window.speechSynthesis.onvoiceschanged = null;
      pickVoiceAndSpeak();
    };
  }
}

function float32ToPcm16(f32) {
  const out = new Int16Array(f32.length);
  for (let i = 0; i < f32.length; i++) out[i] = Math.max(-32768, Math.min(32767, f32[i] * 32768));
  return out;
}
function arrayBufferToBase64(buf) {
  const bytes = new Uint8Array(buf);
  let bin = '';
  bytes.forEach(b => bin += String.fromCharCode(b));
  return btoa(bin);
}
function base64ToArrayBuffer(b64) {
  const bin = atob(b64);
  const buf = new ArrayBuffer(bin.length);
  const view = new Uint8Array(buf);
  for (let i = 0; i < bin.length; i++) view[i] = bin.charCodeAt(i);
  return buf;
}

window.miraSubmitFeedback = async function () {
  const name    = document.getElementById('fb-name').value.trim()    || 'Guest';
  const comment = document.getElementById('fb-comment').value.trim();
  const rating  = window._miraRating || 0;

  if (!comment) { alert('Please enter your feedback.'); return; }
  if (!rating)  { alert('Please select a star rating.'); return; }

  const btn = document.getElementById('fb-submit-btn');
  btn.disabled = true; btn.textContent = 'Submitting…';

  const csrf = document.querySelector('meta[name="csrf-token"]')?.getAttribute('content');
  try {
    const res  = await fetch('/nlah/feedback/submit', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'X-CSRF-TOKEN': csrf },
      body: JSON.stringify({ name, comment, rating })
    });
    const data = await res.json();
    if (data.success) {
      miraHideFeedback();
      appendMsg('ai', 'Salamat sa iyong feedback! 😊 It means a lot to us.');
    } else {
      alert(data.message || 'Could not submit. Please try again.');
      btn.disabled = false; btn.textContent = 'Submit Feedback';
    }
  } catch (_) {
    alert('Network error. Please try again.');
    btn.disabled = false; btn.textContent = 'Submit Feedback';
  }
};
